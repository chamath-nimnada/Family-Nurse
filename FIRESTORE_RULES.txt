// Firebase Firestore Security Rules
// Copy this to Firebase Console → Firestore Database → Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Allow read/write for authenticated users
    match /users/{userId} {
      allow read: if request.auth.uid == userId || 
                     get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'super_admin'];
      allow write: if request.auth.uid == userId || 
                      get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'super_admin';
    }

    // Bookings collection
    match /bookings/{bookingId} {
      // Admins can read/write all bookings
      allow read, write: if get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'super_admin'];
      
      // Nurses can read their assigned bookings
      allow read: if resource.data.assignedNurseId == request.auth.uid;
      
      // Nurses can update status of their assigned bookings
      allow update: if resource.data.assignedNurseId == request.auth.uid &&
                       request.resource.data.status in ['completed', 'cancelled'];
      
      // Public can create bookings (no auth required)
      allow create: if true;
    }

    // Patients collection
    match /patients/{patientId} {
      // Everyone can read
      allow read: if request.auth != null;
      
      // Admins can write
      allow write: if get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'super_admin'];
      
      // Anyone can create
      allow create: if true;
    }

    // Activity logs
    match /activity_logs/{logId} {
      // Admins can read all
      allow read: if get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'super_admin'];
      
      // Authenticated users can create
      allow create: if request.auth != null;
    }

    // Visit logs
    match /visit_logs/{logId} {
      // Nurses can read their own visit logs
      allow read: if resource.data.nurseId == request.auth.uid;
      
      // Admins can read all
      allow read: if get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role in ['admin', 'super_admin'];
      
      // Nurses can create visit logs
      allow create: if resource.data.nurseId == request.auth.uid;
    }

    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
